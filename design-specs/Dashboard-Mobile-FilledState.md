## React Component Sketch

```tsx
interface MobileFilledDashboardProps {
  assignments: Assignment[];
  onSelectAssignment: (id: string) => void;
}

export const MobileFilledDashboard = ({
  assignments,
  onSelectAssignment,
}: MobileFilledDashboardProps) => {
  const [search, setSearch] = useState('');

  const filtered = useMemo(
    () => assignments.filter(a =>
      a.title.toLowerCase().includes(search.toLowerCase())
    ),
    [assignments, search]
  );

  return (
    <div className="mobile-filled-dashboard">
      <MobileStatusBar />
      <MobileHeader />

      <div className="mobile-content">
        <h1 className="mobile-content__title">Assignments</h1>

        <div className="mobile-search-bar">
          <input
            className="mobile-search-input"
            placeholder="Search Name"
            value={search}
            onChange={(e) => setSearch(e.target.value)}
          />
          <button className="mobile-filter-btn">
            <FilterIcon />
            Filter
          </button>
        </div>

        <div className="mobile-assignment-list">
          {filtered.map(a => (
            <article
              key={a.id}
              className="mobile-assignment-card"
              onClick={() => onSelectAssignment(a.id)}
            >
              <h3 className="mobile-assignment-card__title">{a.title}</h3>
              <p className="mobile-assignment-card__meta">{a.subject} • {a.class}</p>
              <span className="mobile-assignment-card__due">Due: {a.dueDate}</span>
            </article>
          ))}
        </div>
      </div>

      <BottomTabBar activeTab="Assignments" />
    </div>
  );
};
```


## Notes & Edge Cases

- **Search on Mobile**: Search input uses a simple text filter. Consider adding a debounce (300ms) to avoid excessive re-renders during typing.
- **Filter Bottom Sheet**: On mobile, "Filter" should open a bottom sheet (not a dropdown) with filter options. Use a library like `react-spring` or `framer-motion` for the sheet animation.
- **Card List**: Assignment cards stack vertically. For very long lists, use virtualized scrolling (`react-window` or `@tanstack/virtual`).
- **Pull to Refresh**: Consider adding pull-to-refresh gesture to reload the assignment list.
- **Desktop Counterpart**: `Filled-State-` (1440×843px).
- **Touch Feedback**: Cards should have `active` state feedback (slight scale or opacity change) on tap.
- **Empty State Transition**: If all assignments are deleted, transition to `Dashboard-Mobile-Empty`.
