## React Component Sketch

```tsx
import { useState, useMemo } from 'react';

// ─── Types ───────────────────────────────────────────────────────────────────
interface Assignment {
  id: string;
  title: string;
  subject: string;
  class: string;
  dueDate: string;
  totalMarks: number;
  totalQuestions: number;
  status: 'draft' | 'published' | 'graded';
  createdAt: string;
}

interface FilledStateProps {
  assignments: Assignment[];
  onCreateNew: () => void;
  onSelectAssignment: (id: string) => void;
}

// ─── Sub-Components ──────────────────────────────────────────────────────────
const SearchFilterBar = ({
  searchQuery,
  onSearchChange,
  filterValue,
  onFilterChange,
}: {
  searchQuery: string;
  onSearchChange: (q: string) => void;
  filterValue: string;
  onFilterChange: (f: string) => void;
}) => (
  <div className="search-filter-row">
    <input
      type="text"
      className="search-input"
      placeholder="Search Assignment"
      value={searchQuery}
      onChange={(e) => onSearchChange(e.target.value)}
    />
    <button className="filter-button">
      <FilterIcon /> Filter By
    </button>
    <select
      className="classes-dropdown"
      value={filterValue}
      onChange={(e) => onFilterChange(e.target.value)}
    >
      <option value="">All Classes</option>
      <option value="5A">Class 5A</option>
      <option value="8B">Class 8B</option>
    </select>
  </div>
);

const AssignmentCard = ({
  assignment,
  onClick,
}: {
  assignment: Assignment;
  onClick: () => void;
}) => (
  <article className="assignment-card" onClick={onClick} role="button" tabIndex={0}>
    <h3 className="assignment-card__title">{assignment.title}</h3>
    <p className="assignment-card__meta">
      {assignment.subject} • {assignment.class}
    </p>
    <div className="assignment-card__footer">
      <span className="assignment-card__due">Due: {assignment.dueDate}</span>
      <span className="assignment-card__marks">
        {assignment.totalMarks} marks • {assignment.totalQuestions} questions
      </span>
    </div>
    <span className={`assignment-card__status assignment-card__status--${assignment.status}`}>
      {assignment.status}
    </span>
  </article>
);

// ─── Main Component ──────────────────────────────────────────────────────────
export const FilledState = ({
  assignments,
  onCreateNew,
  onSelectAssignment,
}: FilledStateProps) => {
  const [searchQuery, setSearchQuery] = useState('');
  const [classFilter, setClassFilter] = useState('');

  const filtered = useMemo(() => {
    return assignments.filter(a => {
      const matchesSearch = a.title.toLowerCase().includes(searchQuery.toLowerCase());
      const matchesClass = !classFilter || a.class === classFilter;
      return matchesSearch && matchesClass;
    });
  }, [assignments, searchQuery, classFilter]);

  return (
    <div className="assignments-dashboard">
      <TopNavBar notificationCount={32} />
      <Sidebar activeRoute="Assignments" />

      <main className="main-content">
        <section className="content-header">
          <div>
            <h1 className="content-header__title">Assignments</h1>
            <p className="content-header__subtitle">
              Manage and create assignments for your classes.
            </p>
          </div>
        </section>

        <SearchFilterBar
          searchQuery={searchQuery}
          onSearchChange={setSearchQuery}
          filterValue={classFilter}
          onFilterChange={setClassFilter}
        />

        <div className="assignment-cards">
          {filtered.map(a => (
            <AssignmentCard
              key={a.id}
              assignment={a}
              onClick={() => onSelectAssignment(a.id)}
            />
          ))}
        </div>
      </main>
    </div>
  );
};
```


## Notes & Edge Cases

- **Search Behavior**: Real-time filtering with debounce (300ms recommended). Searches by assignment title. Consider adding subject/class search.
- **Filter Dropdown**: "Filter By" opens a dropdown or popover with filter options (status, date range, class). Multiple filters can be combined.
- **Classes Dropdown**: Separate dropdown to filter by class. Should show all classes the teacher has created assignments for.
- **Card Interactions**: Each assignment card is clickable, navigating to the assignment detail/output view. Cards need hover state (shadow elevation) and focus state for keyboard navigation.
- **Empty Search Results**: If search/filter returns no results, show an inline empty state ("No assignments match your search").
- **Responsive**: Desktop variant at 1440px. Mobile equivalent is `Dashboard-Mobile-FilledState` (393px).
- **Badge Counts**: AI Teacher's Toolkit badge shows "10" — this is a notification count that updates dynamically.
- **Sidebar State**: "Assignments" nav item is in active state (highlighted background, bold text).
- **Hover/Focus States**: Not in static export. Cards: elevate shadow on hover. Buttons: darken bg on hover. Inputs: border color change on focus.
- **Pagination/Infinite Scroll**: Design shows multiple cards — consider pagination or virtual scrolling for large assignment lists.
- **Accessibility**: Cards need `role="article"` or `role="button"`, keyboard navigation (Tab between cards, Enter to open), and `aria-label` with assignment title.
